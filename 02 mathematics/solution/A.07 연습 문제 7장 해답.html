\[\begin{split} \begin{align} \begin{aligned} S &= \dfrac{1}{N}\sum_{i=1}^N (x_i-\bar{x})(x_i-\bar{x})^T \\ &= \dfrac{1}{N} \left( (x_1-\bar{x})(x_1-\bar{x})^T + \cdot + x_N-\bar{x})(x_N-\bar{x})^T \right) \\ &= \dfrac{1}{N} \begin{bmatrix} x_1-\bar{x} & \cdots & x_N-\bar{x} \end{bmatrix} \begin{bmatrix} (x_1-\bar{x})^T \\ \vdots \\ (x_N-\bar{x})^T \end{bmatrix} \\ &= \dfrac{1}{N} \Big( \begin{bmatrix} x_1 & \cdots & x_N \end{bmatrix} - \begin{bmatrix} \bar{x} & \cdots & \bar{x} \end{bmatrix} \Big) \left( \begin{bmatrix} x_1^T \\ \vdots \\ x_M^T \end{bmatrix} - \begin{bmatrix} \bar{x}^T \\ \vdots \\ \bar{x}^T \end{bmatrix} \right) \\ &= \dfrac{1}{N} (X^T - \bar{x}\mathbf{1_N}^T) (X - \mathbf{1_N}\bar{x}^T) \\ &= \dfrac{1}{N} X_0^TX_0 \end{aligned} \tag{A.7.4} \end{align} \end{split}\]
